int busquedaLineal(Empleado vector[], int tamanio, int valor){
    int i=0;
    int enc=0; // Indica si se ha encontrado el valor */
    int pos=-1; // posición del valor buscado o -1 si no existe
    while (enc==0 && i< tamanio){     
    if(vector[i].cod==valor){
    enc=1;
    pos=i;
    }
    else i++;
    }
    return pos;
    }
